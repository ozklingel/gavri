// ═══════════════════════════════════════ // auth.gs — אימות משתמשים + הרשאות // ═══════════════════════════════════════ const ROLES_SHEET = 'פרטים אישיים'; const REPORT_SHEET = 'דוח 1'; function login(username, password, groupName) { const spreadsheetId = GROUPS[groupName]; if (!spreadsheetId) throw new Error('קבוצה לא נמצאה: ' + groupName); const ss = SpreadsheetApp.openById(spreadsheetId); // ── גיליון נתונים ── const dataSheet = ss.getSheetByName(SHEET_NAME); if (!dataSheet) throw new Error('גיליון "' + SHEET_NAME + '" לא נמצא'); const table = dataSheet.getDataRange().getDisplayValues(); const header = table[0]; const rows = table.slice(1); const inputUser = (username || '').trim(); const inputPass = (password || '').trim(); const matchedRow = rows.find(row => (row[0] || '').trim() === inputUser && (row[1] || '').trim() === inputPass ); if (!matchedRow) return { success: false }; // ── בדיקת הרשאת מפקד ── const rolesSheet = ss.getSheetByName(ROLES_SHEET); if (!rolesSheet) throw new Error('גיליון "' + ROLES_SHEET + '" לא נמצא'); const rolesData = rolesSheet.getDataRange().getDisplayValues(); const userRoleRow = rolesData.find(row => (row[0] || '').trim() === inputUser); const isCommander = userRoleRow ? (userRoleRow[2] || '').trim() === 'מפקד' : false; const driversData = getDrivers(ss); // ── דוח 1 — רק למפקד ── let reportData = null; if (isCommander) { reportData = _getTodayReport(ss); } return { success: true, isCommander: isCommander, formLink: '', header: header, myRow: matchedRow, commanderRows: isCommander ? rows.slice(0, 5) : [], reportData: reportData, drivers: driversData // 🔥 זה מה שחסר לך }; } function getDrivers(ss) { const sheet = ss.getSheetByName('פרטים אישיים'); const data = sheet.getDataRange().getDisplayValues(); const drivers = []; const commanders = []; data.forEach(row => { const name = (row[0] || '').trim(); const role = (row[2] || '').trim(); if (role.includes('נהג')) { drivers.push(name); } if (role === 'מפקד') { commanders.push(name); } }); return { drivers, commanders }; } // ── שולף את עמודות היום ומחר מגיליון דוח 1 ── function _getTodayReport(ss) { const sheet = ss.getSheetByName(REPORT_SHEET); if (!sheet) return null; const todayLabel = _formatDateHebrew(0); const tomorrowLabel = _formatDateHebrew(1); // שימוש ב-getLastRow/getLastColumn כדי לא לפספס שורות const lastRow = sheet.getLastRow(); const lastCol = sheet.getLastColumn(); if (lastRow < 2 || lastCol < 1) return null; const allData = sheet.getRange(1, 1, lastRow, lastCol).getDisplayValues(); const headerRow = allData[0]; const todayCol = headerRow.findIndex(cell => cell.trim() === todayLabel); const tomorrowCol = headerRow.findIndex(cell => cell.trim() === tomorrowLabel); if (todayCol === -1) return null; // שורה 0 = תאריך, שורה 1 = יום בשבוע — מתחילים משורה 2 // לולאה נפרדת לכל עמודה כדי לא לפספס שורות const todayValues = []; for (let i = 2; i < allData.length; i++) { const cell = (allData[i][todayCol] || '').trim(); if (cell !== '') todayValues.push(cell); } const tomorrowValues = []; if (tomorrowCol !== -1) { for (let i = 2; i < allData.length; i++) { const cell = (allData[i][tomorrowCol] || '').trim(); if (cell !== '') tomorrowValues.push(cell); } } // שמות מסבב נוכחי — ערך שונה מ-0 ו-1 const todaySpecial = _getSababNames(ss, todayLabel); const tomorrowSpecial = _getSababNames(ss, tomorrowLabel); return { todayLabel: todayLabel, tomorrowLabel: tomorrowLabel, todayValues: todayValues, tomorrowValues: tomorrowValues, todaySpecial: todaySpecial, tomorrowSpecial: tomorrowSpecial }; } // ── שולף שמות מגיליון סבב נוכחי שהערך שלהם שונה מ-0 ו-1 ── function _getSababNames(ss, dateLabel) { const sheet = ss.getSheetByName('סבב נוכחי'); if (!sheet) return []; const lastRow = sheet.getLastRow(); const lastCol = sheet.getLastColumn(); if (lastRow < 2) return []; const data = sheet.getRange(1, 1, lastRow, lastCol).getDisplayValues(); const headerRow = data[0]; const colIndex = headerRow.findIndex(cell => cell.trim() === dateLabel); if (colIndex === -1) return []; const names = []; for (let i = 7; i < data.length; i++) { const name = (data[i][0] || '').trim(); const val = (data[i][colIndex] || '').trim(); if (name === '') continue; if (val !== '0' && val !== '1' && val !== '') { names.push(name + ' — ' + val); } } return names; } // ── פורמט תאריך כמו בגיליון: "20-מרץ" ── function _formatDateHebrew(offset) { const MONTHS = ['ינו׳','פבר׳','מרץ','אפר׳','מאי','יוני','יולי','אוג׳','ספט׳','אוק׳','נוב׳','דצמ׳']; const d = new Date(); d.setDate(d.getDate() + offset); return d.getDate() + '-' + MONTHS[d.getMonth()]; }
